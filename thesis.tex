\documentclass[a4paper,12pt]{report}


% ******************************** Settings ********************************

\include{Classes/cfg}        % Essential LaTeX packages and macros
\include{thesis-info}        % Thesis/dissertation information
\include{Preamble/preamble}  % Custom LaTeX packages

\begin{document}

% ******************************** Front Matter ********************************
  
  % If you want to remove one of the front matter pages just comment out the

  \maketitle                                 % Create the title page
  \makesignature                             % Create the signature page
  \makelibrarycatalog                        % Insert the library catalog page
  \makeapproval                              % Create the approval page
  \include{Dedication/dedication}            % Insert the dedication page

  \include{Abstract/abstract_pt}             % Insert the abstract page in portuguese
  \include{Abstract/abstract}                % Insert the abstract page in english

  \include{Chapter1/chapter1}
  \include{Chapter2/chapter2}
  \include{Chapter3/chapter3}

  \cleardoublepage
  \printbib         % Insert the bibliography - Change the references path at

  \include{Appendix/A/appendix}  % Insert the appendix A
